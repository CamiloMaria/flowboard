    Create `apps/web/src/components/demo/DemoBanner.tsx`:

    ```typescript
    import { useState } from 'react';
    import { Link } from 'react-router';
    import { motion, AnimatePresence } from 'motion/react';
    import { X } from 'lucide-react';

    const STORAGE_KEY = 'flowboard-demo-banner-dismissed';

    export function DemoBanner() {
      const [dismissed, setDismissed] = useState(() =>
        sessionStorage.getItem(STORAGE_KEY) === 'true'
      );

      function handleDismiss() {
        setDismissed(true);
        sessionStorage.setItem(STORAGE_KEY, 'true');
      }

      return (
        <AnimatePresence>
          {!dismissed && (
            <motion.div
              initial={{ height: 0, opacity: 0 }}
              animate={{ height: 40, opacity: 1 }}
              exit={{ height: 0, opacity: 0 }}
              transition={{ duration: 0.2, ease: [0.16, 1, 0.3, 1] }}
              className="fixed top-0 left-0 right-0 z-50 h-[40px] bg-bg-elevated border-l-[3px] border-l-accent flex items-center justify-between px-4"
            >
              <p className="font-body text-sm text-text-secondary">
                You're watching a live demo — real-time collaboration powered by WebSockets & CRDTs.{' '}
                <Link to="/register" className="text-accent hover:underline font-medium">
                  Sign Up
                </Link>
              </p>
              <button
                onClick={handleDismiss}
                className="text-text-muted hover:text-text-primary transition-colors p-1"
                aria-label="Dismiss demo banner"
              >
                <X size={16} />
              </button>
            </motion.div>
          )}
        </AnimatePresence>
      );
    }
    ```

    Per D-11: Single "Sign Up" CTA, no "Try Demo" button since the guest IS in the demo.
    Per D-13: sessionStorage for dismiss persistence — same session, comes back on new tabs/sessions.
    Per DESIGN.md: fixed top, full-width, `bg-elevated` + `accent` left border, 40px height.
