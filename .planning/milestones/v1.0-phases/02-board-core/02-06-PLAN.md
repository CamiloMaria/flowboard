1. Create `apps/web/src/hooks/useBoardSocket.ts`:
   ```typescript
   import { useEffect } from 'react';
   import { connectSocket, disconnectSocket } from '../lib/socket';
   import { getQueryClient } from '../providers/QueryProvider';
   import { useBoardStore } from '../stores/board.store';
   import type { BoardWithLists, CardMovePayload, CardCreatePayload, CardUpdatePayload, CardDeletePayload, ListCreatePayload, ListUpdatePayload, ListDeletePayload } from '@flowboard/shared';

   export function useBoardSocket(boardId: string) {
     useEffect(() => {
       const socket = connectSocket();
       const queryClient = getQueryClient();
       const { setConnectionStatus } = useBoardStore.getState();

       // Connection status handlers (D-16)
       socket.on('connect', () => {
         setConnectionStatus('connected');
         socket.emit('board:join', { boardId });
       });
       socket.on('disconnect', () => setConnectionStatus('disconnected'));
       socket.on('reconnect_attempt', () => setConnectionStatus('reconnecting'));
       socket.on('reconnect', () => {
         setConnectionStatus('connected');
         socket.emit('board:join', { boardId });
         // Re-sync board data after reconnect
         queryClient.invalidateQueries({ queryKey: ['board', boardId] });
       });

       // If already connected, join room immediately
       if (socket.connected) {
         setConnectionStatus('connected');
         socket.emit('board:join', { boardId });
       }

       // Event handlers (D-13: apply changes directly to TanStack Query cache)
       socket.on('card:move', (payload: CardMovePayload) => {
         queryClient.setQueryData<BoardWithLists>(['board', boardId], (old) => {
           if (!old) return old;
           const lists = old.lists.map(list => ({
             ...list,
             cards: list.cards.filter(c => c.id !== payload.cardId),
           }));
           // Add card to target list
           const targetList = lists.find(l => l.id === payload.card.listId);
           if (targetList) {
             targetList.cards.push(payload.card);
             targetList.cards.sort((a, b) => a.position - b.position);
           }
           return { ...old, lists };
         });
       });

       socket.on('card:create', (payload: CardCreatePayload) => {
         queryClient.setQueryData<BoardWithLists>(['board', boardId], (old) => {
           if (!old) return old;
           return {
             ...old,
             lists: old.lists.map(list =>
               list.id === payload.card.listId
                 ? { ...list, cards: [...list.cards, payload.card].sort((a, b) => a.position - b.position) }
                 : list
             ),
           };
         });
       });

       socket.on('card:update', (payload: CardUpdatePayload) => {
         queryClient.setQueryData<BoardWithLists>(['board', boardId], (old) => {
           if (!old) return old;
           return {
             ...old,
             lists: old.lists.map(list => ({
               ...list,
               cards: list.cards.map(c => c.id === payload.card.id ? { ...c, ...payload.card } : c),
             })),
           };
         });
       });

       socket.on('card:delete', (payload: CardDeletePayload) => {
         queryClient.setQueryData<BoardWithLists>(['board', boardId], (old) => {
           if (!old) return old;
           return {
             ...old,
             lists: old.lists.map(list => ({
               ...list,
               cards: list.cards.filter(c => c.id !== payload.cardId),
             })),
           };
         });
       });

       socket.on('list:create', (payload: ListCreatePayload) => {
         queryClient.setQueryData<BoardWithLists>(['board', boardId], (old) => {
           if (!old) return old;
           return {
             ...old,
             lists: [...old.lists, { ...payload.list, cards: [] }].sort((a, b) => a.position - b.position),
           };
         });
       });

       socket.on('list:update', (payload: ListUpdatePayload) => {
         queryClient.setQueryData<BoardWithLists>(['board', boardId], (old) => {
           if (!old) return old;
           return {
             ...old,
             lists: old.lists.map(l => l.id === payload.list.id ? { ...l, ...payload.list } : l),
           };
         });
       });

       socket.on('list:delete', (payload: ListDeletePayload) => {
         queryClient.setQueryData<BoardWithLists>(['board', boardId], (old) => {
           if (!old) return old;
           return {
             ...old,
             lists: old.lists.filter(l => l.id !== payload.listId),
           };
         });
       });

       // Cleanup
       return () => {
         socket.emit('board:leave', { boardId });
         socket.off('connect');
         socket.off('disconnect');
         socket.off('reconnect_attempt');
         socket.off('reconnect');
         socket.off('card:move');
         socket.off('card:create');
         socket.off('card:update');
         socket.off('card:delete');
         socket.off('list:create');
         socket.off('list:update');
         socket.off('list:delete');
         disconnectSocket();
       };
     }, [boardId]);
   }
   ```
