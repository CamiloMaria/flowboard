    **Enhanced health endpoint:** Update `apps/api/src/app.service.ts` to inject PrismaService and Redis (via `@Inject('REDIS_CLIENT')`) and check connectivity:
    ```typescript
    import { Injectable, Inject } from '@nestjs/common';
    import { PrismaService } from './prisma/prisma.service';
    import Redis from 'ioredis';

    @Injectable()
    export class AppService {
      constructor(
        private readonly prisma: PrismaService,
        @Inject('REDIS_CLIENT') private readonly redis: Redis,
      ) {}

      async getHealth() {
        const checks = { database: 'disconnected', redis: 'disconnected' };
        try {
          await this.prisma.$queryRaw`SELECT 1`;
          checks.database = 'connected';
        } catch {}
        try {
          await this.redis.ping();
          checks.redis = 'connected';
        } catch {}
        const allHealthy = checks.database === 'connected' && checks.redis === 'connected';
        return {
          status: allHealthy ? 'ok' : 'error',
          ...checks,
          timestamp: new Date().toISOString(),
        };
      }
    }
    ```

    Update `apps/api/src/app.controller.ts` to add `/health` route (in addition to existing root `/`):
    ```typescript
    @Public()
    @Get('health')
    async getHealth() {
      const health = await this.appService.getHealth();
      // Return 503 if unhealthy
      if (health.status !== 'ok') {
        throw new HttpException(health, 503);
      }
      return health;
    }
    ```

    Note: The controller has `@Controller()` with no prefix, but `main.ts` sets `app.setGlobalPrefix('api')`, so `/health` → `/api/health`. Keep the existing root GET handler as well.

| Docker API → PostgreSQL/Redis | Internal bridge network, no external exposure |
| .env file on server | Secrets stored in plaintext on VM filesystem |
